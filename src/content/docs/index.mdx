  // ── Fetch releases with caching ──
  const cache = {};
  async function getRelease(repo) {
    if (cache[repo]) return cache[repo];
    try {
      const res = await fetch('https://api.github.com/repos/' + repo + '/releases/latest');
      if (!res.ok) return null;
      const data = await res.json();
      cache[repo] = data;
      return data;
    } catch (e) { return null; }
  }

  // ── Deduplicate repos and batch fetch ──
  const els = document.querySelectorAll('.platform-downloads');
  const repos = [...new Set(Array.from(els).map(el => el.dataset.repo).filter(Boolean))];

  Promise.all(repos.map(r => getRelease(r))).then(() => {
    els.forEach(el => {
      const repo = el.dataset.repo;
      if (!repo || !cache[repo]) return;
      const assets = cache[repo].assets || [];
      const win = assets.find(a => /setup.*\\.exe$/i.test(a.name)) || assets.find(a => /\\.exe$/i.test(a.name)) || assets.find(a => /\\.msi$/i.test(a.name));
      const mac = assets.find(a => /arm64.*\\.dmg$/i.test(a.name)) || assets.find(a => /\\.dmg$/i.test(a.name));
      const linux = assets.find(a => /\\.AppImage$/i.test(a.name)) || assets.find(a => /\\.deb$/i.test(a.name));
      if (!win && !mac && !linux) return;
      let html = '';
      if (win) html += '<a href="' + win.browser_download_url + '" class="dl-icon" title="Download for Windows">' + icons.win + '</a>';

      if (linux) html += '<a href="' + linux.browser_download_url + '" class="dl-icon" title="Download for Linux">' + icons.linux + '</a>';
      el.innerHTML = html;
    });
  });

  // ── Carousel dots ──
  const carousel = document.querySelector('.carousel');
  const cards = document.querySelectorAll('.carousel-card');
  const dotsContainer = document.getElementById('carousel-dots');
  if (carousel && dotsContainer && cards.length) {
    cards.forEach((_, i) => {
      const dot = document.createElement('button');
      dot.className = 'dot' + (i === 0 ? ' active' : '');
      dot.setAttribute('aria-label', 'Go to slide ' + (i + 1));
      dot.addEventListener('click', () => {
        cards[i].scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' });
      });
      dotsContainer.appendChild(dot);
    });
    const dots = dotsContainer.querySelectorAll('.dot');
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const idx = Array.from(cards).indexOf(entry.target);
          dots.forEach((d, i) => d.classList.toggle('active', i === idx));
        }
      });
    }, { root: carousel, threshold: 0.6 });
    cards.forEach(card => observer.observe(card));
  }
});
